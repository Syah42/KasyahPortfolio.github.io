    <script>
        let currentSongIndex = 0;
        const songs = document.querySelectorAll('.song');
        const audios = document.querySelectorAll('audio');
        const playPauseBtn = document.getElementById('playPauseBtn');
        let isPlaying = false;

        function playPauseSong() {
            const currentAudio = audios[currentSongIndex];
            if (isPlaying) {
                currentAudio.pause();
                isPlaying = false;
                playPauseBtn.innerHTML = '<i class="fas fa-play"></i> Play';
            } else {
                currentAudio.play();
                isPlaying = true;
                playPauseBtn.innerHTML = '<i class="fas fa-pause"></i> Pause';
            }
        }

        function nextSong() {
            // Stop the current song
            const currentAudio = audios[currentSongIndex];
            currentAudio.pause();
            currentAudio.currentTime = 0; // Reset to the beginning

            // Hide the current song
            songs[currentSongIndex].classList.remove('active');

            // Increment the song index
            currentSongIndex = (currentSongIndex + 1) % songs.length; // Loop back to the first song

            // Show the next song
            songs[currentSongIndex].classList.add('active');

            // Update play button and play the next song
            isPlaying = true;
            playPauseBtn.innerHTML = '<i class="fas fa-pause"></i> Pause';
            audios[currentSongIndex].play();
        }
    </script>
